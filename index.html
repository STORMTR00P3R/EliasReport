  • Reverse Integer
    • Problem Statement: Given a signed 32-bit integer x, return x with its digits reversed. If reversing x causes the value to go outside the signed 32-bit integer range [-231, 231 - 1], then return 0. 
    • 
                                class Solution {
                                  func reverse(_ x: Int) -> Int {
                                      var result = 0
                                      var number = x

                                      while number != 0 {

                                        let digit = number % 10
                                        number /= 10

                                        // Check for integer overlow

                                        if result > Int32.max / 10 || (result == Int32.max / 10 && digit > 7) {
                                          return 0
                                        }

                                        if result < Int32.min / 10 || (result == Int32.min / 10 && digit < -8) {
                                          return 0
                                        }


                                        result = result * 10 + digit


                                      }
                                    return result
                                  }
                              }
  • Roman to Integer
    • Problem Statement: Given a roman numeral, convert it to an integer. 
    • 
                                    class Solution {
                                      func romanToInt(_ s: String) -> Int {

                                        let numerals: [Character: Int] = [
                                          "I": 1,
                                          "V": 5,
                                          "X": 10,
                                          "L": 50,
                                          "C": 100,
                                          "D": 500,
                                          "M": 1000
                                        ]

                                        var result = 0
                                        var prevValue = 0

                                        for char in s.reversed() {
                                          if let currentInt = numerals[char] {
                                            if currentInt < prevValue {
                                              result -= currentInt
                                            } else {
                                              result += currentInt
                                            }
                                            prevValue = currentInt
                                          }
                                        }


                                        return result
                                      }
                                  }
  • Palindrome Number
    • Problem Statement: Given an integer x, return true if x is a palindrome, and false otherwise. 
    • 
                                        class Solution {
                                          func isPalindrome(_ x: Int) -> Bool {
                                              if (x < 0) {
                                                return false
                                              }

                                              var original = x
                                              var reversed = 0

                                              while original > 0 {
                                                reversed = reversed * 10 + original % 10
                                                original /= 10
                                              }

                                              return reversed == x

                                          }
                                      }
  • Valid Parentheses
    • Problem Statement: Given a string s containing just the characters '(', ')', '{', '}', '[' and ']', determine if the input string is valid. 
    • 
                                    class Solution {
                                        func isValid(_ s: String) -> Bool {

                                            var openCount = 0
                                            var curlyCount = 0
                                            var squareCount = 0

                                            for char in s {
                                              switch char {

                                                case "(":
                                                    openCount += 1
                                                case ")":
                                                    if openCount == 0 {
                                                        return false
                                                    }
                                                    openCount -= 1

                                                case "{":
                                                    curlyCount += 1  
                                                case "}":
                                                    if curlyCount == 0 {
                                                      return false
                                                  }
                                                    curlyCount -= 1

                                                case "[":
                                                    squareCount += 1
                                                case "]":
                                                  if squareCount == 0 {
                                                    return false
                                                  }
                                                  squareCount -= 1


                                                default:
                                                    return false
                                              }
                                            }

                                              return openCount == 0 && curlyCount == 0 && squareCount == 0
                                        }
                                    }

    • Problem Statement: Given an integer array nums and an integer val, remove all occurrences of val in nums in-place. The order of the elements may be changed. Then return the number of elements in nums which are not equal to val. 
    • 
                                          class Solution {
                                            func removeElement(_ nums: inout [Int], _ val: Int) -> Int {
                                                var i = 0

                                                for j in 0.. < nums.count {
                                                    if nums[j] != val {
                                                      nums[i] = nums[j]
                                                      i += 1
                                                    }
                                                }
                                                return i
                                            }
                                        }
  • Plus One
    • Problem Statement: You are given a large integer represented as an integer array digits, where each digits[i] is the ith digit of the integer. The digits are ordered from most significant to least significant in left-to-right order. The large integer does not contain any leading 0's.
    •                         class Solution {
                            func plusOne(_ digits: [Int]) -> [Int] {
                                var result = digits 
                                var carry = 1

                                for i in stride(from: digits.count - 1, through: 0, by: -1) {
                                    let sum = digits[i] + carry
                                    result[i] = sum % 10
                                    carry = sum / 10
                                }

                                if carry > 0 {
                                    result.insert(carry, at: 0)
                                }

                                return result
                            }
                        }

  • Two Sum
    • Problem Statement: Given an array of integers nums and an integer target, return indices of the two numbers such that they add up to target. You may assume that each input would have exactly one solution, and you may not use the same element twice. You can return the answer in any order. 
    • 
                          class Solution {
                              func twoSum(_ nums: [Int], _ target: Int) -> [Int] {
                                  var numIndices = [Int: Int]()
        
                                  for (index, number) in nums.enumerated() {
                                      let complement = target - number
        
                                      if let complementIndex = numIndices[complement] {
                                          return [complementIndex, index]
                                      }
        
                                      numIndices[number] = index
                                  }
        
                                  return []
                              }
                          }
